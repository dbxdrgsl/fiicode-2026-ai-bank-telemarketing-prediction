## exp062: Residual Correction Network

**Idea:** Train neural network to learn what exp058 got WRONG.

```bash
python3 << 'TRAIN'
import torch, torch.nn as nn
import numpy as np, pandas as pd
from sklearn.metrics import roc_auc_score

# Load exp058 OOF
exp058 = pd.read_csv('outputs/oof/exp058_blend_exp052_catneural/oof_predictions.csv')
y_true = exp058['y_true'].values
y_pred = exp058['oof_pred'].values

# Residuals
residuals = y_true - y_pred

# Build correction network
class Corrector(nn.Module):
    def __init__(self):
        super().__init__()
        self.fc1 = nn.Linear(1, 64)
        self.fc2 = nn.Linear(64, 32)
        self.fc3 = nn.Linear(32, 1)
    
    def forward(self, x):
        return self.fc3(torch.relu(self.fc2(torch.relu(self.fc1(x)))))

model = Corrector().cuda()
optimizer = torch.optim.Adam(model.parameters(), lr=0.01)
criterion = nn.MSELoss()

X = torch.tensor(y_pred, dtype=torch.float32).view(-1,1).cuda()
y = torch.tensor(residuals, dtype=torch.float32).view(-1,1).cuda()

for epoch in range(200):
    optimizer.zero_grad()
    pred = model(X)
    loss = criterion(pred, y)
    loss.backward()
    optimizer.step()

# Corrected OOF
with torch.no_grad():
    corrections = model(X).cpu().numpy().squeeze()
    corrected = np.clip(y_pred + 0.3 * corrections, 0, 1)
    
auc = roc_auc_score(y_true, corrected)
print(f"Corrected OOF AUC: {auc:.6f}")

# Save
import json
from pathlib import Path
Path('outputs/oof/exp062_exp058_residual_correction').mkdir(parents=True, exist_ok=True)
Path('outputs/submissions/exp062_exp058_residual_correction').mkdir(parents=True, exist_ok=True)
Path('outputs/logs/exp062_exp058_residual_correction').mkdir(parents=True, exist_ok=True)

pd.DataFrame({
    'id': exp058['id'],
    'y_true': y_true,
    'oof_pred': corrected
}).to_csv('outputs/oof/exp062_exp058_residual_correction/oof_predictions.csv', index=False)

# For test: load exp058 test, apply same correction
exp058_test = pd.read_csv('outputs/submissions/exp058_blend_exp052_catneural/submission.csv')
test_X = torch.tensor(exp058_test['Subscribed'].values, dtype=torch.float32).view(-1,1).cuda()
with torch.no_grad():
    test_corrections = model(test_X).cpu().numpy().squeeze()
    test_corrected = np.clip(exp058_test['Subscribed'].values + 0.3 * test_corrections, 0, 1)

pd.DataFrame({
    'id': exp058_test['id'],
    'Subscribed': test_corrected
}).to_csv('outputs/submissions/exp062_exp058_residual_correction/submission.csv', index=False)

json.dump({'oof_auc': float(auc), 'method': 'residual_correction'}, 
          open('outputs/logs/exp062_exp058_residual_correction/best_run_summary.json', 'w'))

TRAIN
```

**Expected:** OOF improvement if residuals are learnable
**Submit if:** OOF > 0.938200
